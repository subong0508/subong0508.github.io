### [백준 11497번](https://www.acmicpc.net/problem/11497)

**풀이**

```python3
t = int(input())


def solution(n, arr):
    ans = [-1] * n
    arr.sort(reverse=True)
    l, r = 0, n-1
    i = 0
    while arr:
        x = arr.pop()
        if i % 2 == 0:
            ans[l] = x
            l += 1
        else:
            ans[r] = x 
            r -= 1
        i += 1
    max_len = 0
    for i in range(n):
        max_len = max(max_len, abs(ans[i]-ans[i-1]))
    return max_len   


res = [] 
for _ in range(t):
    n = int(input())
    arr = list(map(int, input().split()))
    res.append(solution(n, arr))

print(*res, sep='\n')
```
